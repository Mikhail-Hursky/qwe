Inheritance in ES 5

                    
                        function extend(Child, Parent) {
                            Child.prototype = Object.create(Parent.prototype);
                            Child.prototype.constructor = Child;
                            Child.superclass = Parent.prototype; // optional
                        }

This is an object

							
                                class Transformer {...}

                                class Decepticon extends Transformer {
                                    constructor(name, isRun) {
                                        super(name, isRun);
                                    }

                                    transformation() {
                                        console.log(`${this.name} transform plane`);
                                    }
                                }

                                class Autobot extends Transformer {

                                    constructor(name, isRun) {
                                        super(name, isRun);
                                    }

                                    transformation() {
                                        console.log(`${this.name} transform auto`);
                                    }
                                }
							
						
							
                                class Transformer {...}
